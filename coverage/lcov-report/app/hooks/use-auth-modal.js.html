/**

 * @returns {Object} - Object props to be spread on to the AuthModal component

export const useAuthModal = (initialView = LOGIN_VIEW) => {
    const {isOpen, onOpen, onClose} = useDisclosure()
 
    return {
        initialView,
        isOpen,
        onOpen,
